1function output= ekf_localization_sim(lm, wp)
  2  
  3% INPUTS: 
  4%   lm - set of landmarks
  5%   wp - set of waypoints
  6%
  7% OUTPUTS:
  8%   data - a data structure containing:
  9%          data.true: the vehicle 'true'-path (ie, where the vehicle *actually* went)
 10%          data.path: the vehicle path estimate (ie, estimates traversed vehicle path)
 11%          data.state(k).x: the state vector at time k
 12%          data.state(k).P: the diagonals of the covariance matrix at time k
 13 
 14format compact
 15configfile; % ** USE THIS FILE TO CONFIGURE THE EKF-Localization**
 16
 17
 18% Some boring stuff to setup plots :(but helps in getting good visuualization)
 19
 20fig1=figure;
 21plot(lm(1,:),lm(2,:),'b*')
 22hold on, axis([-180 150 -120 150])
 23plot(wp(1,:),wp(2,:), 'g', wp(1,:),wp(2,:),'g.')
 24xlabel('metres'), ylabel('metres')
 25title('Simulated Environment')
 26
 27fig=figure;
 28plot(lm(1,:),lm(2,:),'b*')
 29hold on,  axis([-180 150 -120 150])
 30xlabel('metres'), ylabel('metres')
 31title('EKF-Localization')
 32
 33h= setup_animations;
 34veh= [0 -WHEELBASE -WHEELBASE; 0 -2 2]; % vehicle animation
 35plines=[]; % for laser line animation
 36pcount=0;
 37
 38% initialise states
 39xtrue= zeros(3,1);
 40x= zeros(3,1);
 41P= zeros(3);
 42
 43% Initialise other variables and constants
 44
 45dt= DT_CONTROLS;    % Time interval between control signals in 'seconds'
 46dtsum= 0;           % change in time since last observation
 47ftag= 1:size(lm,2); % identifier for each landmark
 48 
 49iwp = 1; % index to first waypoint 
 50G = 0; % initial steer angle
 51output = initialise_store(x,P,xtrue); % stored data for off-line
 52QE= Q; RE= R;  
 53 
 54l=2;
 55E=0;
 56Pd=zeros(3,5590);
 57
 58% main loop 
 59while iwp ~= 0
 60    
 61    % compute true data
 62    [G,iwp]= compute_steering(xtrue, wp, iwp, AT_WAYPOINT, G, RATEG, MAXG, dt);
 63    %if iwp==0 & NUMBER_LOOPS > 1, iwp=1; NUMBER_LOOPS= NUMBER_LOOPS-1; end % perform loops: if final waypoint reached, go back to first
 64    xtrue= vehicle_model(xtrue, V,G,dt);
 65    
 66    % Adding noise to controls.
 67    [Vn,Gn]= add_control_noise(V,G,Q, SWITCH_CONTROL_NOISE);
 68    
 69    % EKF predict step ----------------------------> TO DO
 70    [x,P]= EKF_predict(x,P, Vn,Gn,QE,dt);
 71    dtsum= dtsum + dt;
 72     
 73    if dtsum >= DT_OBSERVE
 74    dtsum = 0;
 75    % generating observations
 76    [z,ftag_visible]= get_observations(xtrue, lm, ftag, MAX_RANGE);
 77    z= add_observation_noise(z,R, SWITCH_SENSOR_NOISE);
 78    
 79    
 80    idf = ftag_visible; % known data association. i.e this idf has indices of the landmarks from which observations are generated.  
 81    
 82    
 83    % EKF update step ------------------------------> TO DO
 84    [x,P]= EKF_update(x,P,z,RE,idf,lm); 
 85    
 86        
 87    end
 88    E=E+(xtrue-x).^2; %error in estimated pose
 89    % offline data store
 90    output= store_data(output, x, P, xtrue);
 91    Pd(:,1)=[0;0;0];
 92    det_P(1)=0;
 93    Pd(:,l)=diag(P);
 94    det_P(l)=det(P);
 95    l=l+1;
 96
 97%  <-----------------------------------------------VISULAIZATION--------------------------------------------->
 98    
 99    % plots to get visualization
100    xt= TransformToGlobal(veh,xtrue);
101    xv= TransformToGlobal(veh,x);
102    set(h.xt, 'xdata', xt(1,:), 'ydata', xt(2,:))
103    set(h.xv, 'xdata', xv(1,:), 'ydata', xv(2,:))
104    
105    ptmp= make_covariance_ellipses(x,P);
106    pcov(:,1:size(ptmp,2))= ptmp;
107%     output.i;
108    if dtsum==0
109        %when only predict
110        %if(output.i < 2500)
111            set(h.cov, 'xdata', pcov(1,:), 'ydata', pcov(2,:)) 
112        %else
113        %    set(h.cov, 'Visible', 'off');
114        %end
115        pcount= pcount+1;
116        if pcount == 15
117            set(h.ptht, 'xdata', output.true(1,1:output.i), 'ydata', output.true(2,1:output.i))
118            set(h.pth, 'xdata', output.path(1,1:output.i), 'ydata', output.path(2,1:output.i))
119            pcount=0;
120        end
121        if ~isempty(z)
122            plines= make_laser_lines (lm,x,idf);
123            set(h.obs, 'xdata', plines(1,:), 'ydata', plines(2,:))
124        end
125    end
126    drawnow
127end
128
129output= finalise_data(output);
130set(h.ptht, 'xdata', output.true(1,:), 'ydata', output.true(2,:)) ;  
131set(h.pth, 'xdata', output.path(1,:), 'ydata', output.path(2,:));
132save('output.mat', 'output');
133
134
135figure;
136plot(output.true(1,:),'-b')
137hold on 
138plot(output.path(1,:),'-r')
139hold on 
140plot(output.path(1,:)+3*sqrt(Pd(1,:)),'-g')
141hold on 
142plot(output.path(1,:)-3*sqrt(Pd(1,:)),'-y')
143title('X-coordinates of pose with time')
144legend('True pose_x','Estimated pose_x','mean+3sigma','mean-3sigma')
145figure;
146plot(output.true(2,:),'-b')
147hold on 
148plot(output.path(2,:),'-r')
149hold on 
150plot(output.path(2,:)+3*sqrt(Pd(2,:)),'-g')
151hold on 
152plot(output.path(2,:)-3*sqrt(Pd(2,:)),'-y')
153title('Y-coordinates of pose with time')
154legend('True pose_y','Estimated pose_y','mean+3sigma','mean-3sigma')
155figure;
156plot(output.true(3,:),'-b')
157hold on 
158plot(output.path(3,:),'-r')
159hold on 
160plot(output.path(3,:)+3*sqrt(Pd(3,:)),'-g')
161hold on 
162plot(output.path(3,:)-3*sqrt(Pd(3,:)),'-y')
163title('theta-coordinates of pose with time')
164legend('True pose_y','Estimated pose_y','mean+3sigma','mean-3sigma')
165figure;
166plot(det_P,'-b')
167title('Determinant of covariance with time')
168error=sqrt(E)
169
